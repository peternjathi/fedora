#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>
#include <fcntl.h>

int main(int argc, char* argv[]) {
    char *input_line = NULL;
    size_t len = 0;
    ssize_t read;

    while (1) {
        // 1. Print the prompt
        printf("%% ");
        
        // 2. Get user input
        read = getline(&input_line, &len, stdin);

        // Handle Ctrl+D (EOF)
        if (read == -1) {
            break;
        }

        // Remove newline character
        if (read > 0 && input_line[read - 1] == '\n') {
            input_line[read - 1] = '\0';
        }

        // Skip empty lines
        if (strlen(input_line) == 0) {
            continue;
        }

        // 3. Tokenize input (split by space)
        char *command = strtok(input_line, " ");
        char *next_token = strtok(NULL, " ");
        char *outfile = NULL;

        // Check for redirection ">"
        if (next_token != NULL && strcmp(next_token, ">") == 0) {
            outfile = strtok(NULL, " "); // Get filename
        }

        // 4. Fork process
        pid_t pid = fork();

        if (pid < 0) {
            perror("fork failed");
            exit(1);
        }
        else if (pid == 0) {
            // --- CHILD PROCESS ---

            // 5. Handle Redirection
            if (outfile != NULL) {
                int fd = open(outfile, O_WRONLY | O_CREAT | O_TRUNC, 0666);
                if (fd < 0) {
                    perror("open failed");
                    exit(1);
                }
                dup2(fd, STDOUT_FILENO); // Redirect stdout to file
                close(fd);
            }

            // 6. Execute command
            execlp(command, command, NULL);
            
            // If execlp returns, it failed
            perror("exec failed");
            exit(1);
        }
        else {
            // --- PARENT PROCESS ---
            waitpid(pid, NULL, 0);
        }
    }

    free(input_line);
    printf("\n");
    return 0;
}
